\begin{frame}
    \frametitle{Модель}
    \begin{itemize}
        \item $I$ -- множество учеников
        \item $S$ -- множество школ
        \item $q_S$ -- количество мест в школе (квота)
        \item $P_i$ -- отношение строгого порядка на множестве $S \cup \left\{i\right\}$ (задаёт предпочтения ученика $i$)
        \item Школа $s$ допустима для ученика $i$, если $s P_i i$.
        \item $R_i$ -- отношение слабого порядка на множестве $I \cup \left\{s\right\}$ (задаёт предпочтения  $s$)
        \item Ученик $i$ допустим для школы $s$, если $i R_s s$.
    \end{itemize}
\end{frame}
